012:     public void drawGrid(Graphics g) {
013:       for (int are = 0; are < 7; are++) {
014:         for (int see = 0; see < 8; see++) {
015:           drawDigitGivenCentre(g, 30 + see * 20, 30 + are * 20, 20,
016:               master.grid[are][see]);
017:         }
018:       }
019:     }
020: 
021:     public void drawGridLines(Graphics g) {
022:       g.setColor(Color.LIGHT_GRAY);
023:       for (int are = 0; are <= 7; are++) {
024:         g.drawLine(20, 20 + are * 20, 180, 20 + are * 20);
025:       }
026:       for (int see = 0; see <= 8; see++) {
027:         g.drawLine(20 + see * 20, 20, 20 + see * 20, 180);
028:       }
029:     }
030: 
031:     public void drawHeadings(Graphics g) {
032:       for (int are = 0; are < 7; are++) {
033:         fillDigitGivenCentre(g, 10, 30 + are * 20, 20, are+1);
034:       }
035: 
036:       for (int see = 0; see < 8; see++) {
037:         fillDigitGivenCentre(g, 30 + see * 20, 10, 20, see+1);
038:       }
039:     }
040: 
041:     public void drawDomino(Graphics g, Domino d) {
042:       if (d.placed) {
043:         int y = Math.min(d.ly, d.hy);
044:         int x = Math.min(d.lx, d.hx);
045:         int w = Math.abs(d.lx - d.hx) + 1;
046:         int h = Math.abs(d.ly - d.hy) + 1;
047:         g.setColor(Color.WHITE);
048:         g.fillRect(20 + x * 20, 20 + y * 20, w * 20, h * 20);
049:         g.setColor(Color.RED);
050:         g.drawRect(20 + x * 20, 20 + y * 20, w * 20, h * 20);
051:         drawDigitGivenCentre(g, 30 + d.hx * 20, 30 + d.hy * 20, 20, d.high,
052:             Color.BLUE);
053:         drawDigitGivenCentre(g, 30 + d.lx * 20, 30 + d.ly * 20, 20, d.low,
054:             Color.BLUE);
055:       }
056:     }
057: 
058:     void drawDigitGivenCentre(Graphics g, int x, int y, int diameter, int n) {
059:       int radius = diameter / 2;
060:       g.setColor(Color.BLACK);
061:       // g.drawOval(x - radius, y - radius, diameter, diameter);
062:       FontMetrics fm = g.getFontMetrics();
063:       String txt = Integer.toString(n);
064:       g.drawString(txt, x - fm.stringWidth(txt) / 2, y + fm.getMaxAscent() / 2);
065:     }
066: 
067:     void drawDigitGivenCentre(Graphics g, int x, int y, int diameter, int n,
068:         Color c) {
069:       int radius = diameter / 2;
070:       g.setColor(c);
071:       // g.drawOval(x - radius, y - radius, diameter, diameter);
072:       FontMetrics fm = g.getFontMetrics();
073:       String txt = Integer.toString(n);
074:       g.drawString(txt, x - fm.stringWidth(txt) / 2, y + fm.getMaxAscent() / 2);
075:     }
076: 
077:     void fillDigitGivenCentre(Graphics g, int x, int y, int diameter, int n) {
078:       int radius = diameter / 2;
079:       g.setColor(Color.GREEN);
080:       g.fillOval(x - radius, y - radius, diameter, diameter);
081:       g.setColor(Color.BLACK);
082:       g.drawOval(x - radius, y - radius, diameter, diameter);
083:       FontMetrics fm = g.getFontMetrics();
084:       String txt = Integer.toString(n);
085:       g.drawString(txt, x - fm.stringWidth(txt) / 2, y + fm.getMaxAscent() / 2);
086:     }
