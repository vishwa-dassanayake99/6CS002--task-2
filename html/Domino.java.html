02: public class Domino implements Comparable<Domino> {
03:   public int high;
04:   public int low;
05:   public int hx;
06:   public int hy;
07:   public int lx;
08:   public int ly;
09:   public boolean placed = false;
10: 
11:   public Domino(int high, int low) {
12:     super();
13:     this.high = high;
14:     this.low = low;
15:   }
16:   
17:   public void place(int hx, int hy, int lx, int ly) {
18:     this.hx = hx;
19:     this.hy = hy;
20:     this.lx = lx;
21:     this.ly = ly;
22:     placed = true;
23:   }
24: 
25:   public String toString() {
26:     StringBuffer result = new StringBuffer();
27:     result.append("[");
28:     result.append(Integer.toString(high));
29:     result.append(Integer.toString(low));
30:     result.append("]");
31:     if(!placed){
32:       result.append("unplaced");
33:     } else {
34:       result.append("(");
35:       result.append(Integer.toString(hx+1));
36:       result.append(",");
37:       result.append(Integer.toString(hy+1));
38:       result.append(")");
39:       result.append("(");
40:       result.append(Integer.toString(lx+1));
41:       result.append(",");
42:       result.append(Integer.toString(ly+1));
43:       result.append(")");
44:     }
45:     return result.toString();
46:   }
47: 
48:   /** turn the domino around 180 degrees clockwise*/
49:   
50:   public void invert() {
51:     int tx = hx;
52:     hx = lx;
53:     lx = tx;
54:     
55:     int ty = hy;
56:     hy = ly;
57:     ly = ty;    
58:   }
59: 
60:   public boolean ishl() {    
61:     return hy==ly;
62:   }
63: 
64: 
65:   public int compareTo(Domino arg0) {
66:     if(this.high < arg0.high){
67:       return 1;
68:     }
69:     return this.low - arg0.low;
70:   }
